        public class EQ {
          private int [][] queens=new int [8][8];
          public EQ () {
              for (int i=0; i<8; i++) {
                  for (int j=0; j<8; j++) {
                      this.queens[i][j] =-1;
                  }
              }
          }
          public EQ (int[] Queens) {
              int j=0;
             for (int i=0; i<8; i++) {
                 int temp=Queens[i];
                 this.queens[j][temp]=1;
                 j++;
             }
         
              for (int i=0; i<8; i++) {
                  for (int v=0; v<8; v++) {
                      if (queens[i][v] != 1) {
                          this.queens[i][v] = -1;
                      }
                  }
              }
         
         
          }
         public int get (int i) {
              return queens[i][0];
         }
         public void set (int i, int j) {
                 queens[i][j]=1 ;
          }
             public void printBoard() {

             public boolean isSolved() {
                 int x=10;
                 int y=10;
                 int a=1;
                 for (int i = 0; i < 8; i++) {
                     for (int j = 0; j < 8; j++) {
                         if (i==0) {
                             if (queens[i][j]==1) {
                                 x=i;
                                 y=j;
                             }
                         }
                         if (queens[i][j]==1) {
                             if (x != 10 & y != 10) {
                                 if (getDistance(i, j, x, y) >= 1) {
                                     a++;
                                 }
                             }
                         }
                         if (queens[i][j]==1) {
                             x=i;
                             y=j;
                         }
         
                     }
         
                 }
                
                 if (a!=8) {
                     return false;
                 }
                 else {
                     return true;
                 }
         
             }
             public double getDistance (int x, int y, int X, int Y) {
                 double a =((X-x)*(X-x)) + ((Y-y)*(Y-y));
                 double b= Math.sqrt (a);
                 return b;
             }
         }
